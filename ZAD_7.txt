SELECT 
    COUNT(oceny.ocena) AS ilosc_ocen,
    gry.nazwa
FROM oceny
JOIN gry ON gry.id_gry = oceny.id_gry
GROUP BY gry.nazwa
ORDER BY ilosc_ocen DESC;

 Round( AVG(oceny.ocena),2) as Srednia_ocen ,
 gry.nazwa 
FROM oceny
